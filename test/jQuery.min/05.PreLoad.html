Description:
PreLoad the sound when add sound source.
Result:
Alert 'Loaded!'.
Code:
            function onload() {
                $('#audio').jWebAudio('addSoundSource', {
                    'url': '../../examples/resource/a.ogg',
                    'preLoad': true,
                    'callback': function() {
                        alert('Loaded!');
                    }
                });
            }
