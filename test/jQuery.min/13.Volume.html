Volume changes with input.
Code:
            function onload() {   
                $('#audio').jWebAudio('addSoundSource', {
                    'url': '../../examples/resource/a.ogg',
                    'preLoad': true,
                    'callback': function() {
                        $('#audio').jWebAudio('play');
                        $('#volume')[0].disabled = false;
                    },
                    'volume': 50
                });
            }
            
            function volumeChange() {
                var vol = $('#volume')[0].value;
                // set volume
                $('#audio').jWebAudio('options', {
                    volume: parseInt(vol)
                });
                // get volume
                $('#output').text($('#audio').jWebAudio('options').volume);
            }
