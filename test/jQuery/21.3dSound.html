Play sound 3D effect using default parameters.
Result:
Note the sound 3D Effect difference.

            function onload() {
                $('#audio').jWebAudio('addSoundSource', {
                    'url': '../../examples/resource/a.ogg',
                    'preLoad': true,
                    'callback': function() {
                        $('#audio').jWebAudio('play');
                    }
                });
            }
            
            var useEffect = false;
            var id = null;
            function toggleEffect() {
                useEffect = !useEffect;
                if (useEffect) {
                    id = $('#audio').jWebAudio('addEffect', '3d');
                } else if (id !== null) {
                    source.sound.deleteEffect(id);
                }
            }
